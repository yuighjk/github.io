换行代码如下：

let success='', fail='';
for(let o in res.data.ToSuccess) {
    success += res.data.ToSuccess[o] + " "
    }
    for(let p in res.data.ToFailure) {
    fail += res.data.ToFailure[p] + " "
    }
    let data = ['转入成功的人：' + success,
                '转入失败的人：' + fail]
    console.log(data)
    let newData = []
    const h = this.$createElement
    for(let i in data){
    newData.push(h('p',null ,data[i]))     
    }
    this.$notify({
    title: '提示',
    message: h('div', null ,newData),
    duration: 0
    });

使用vue的$createElement方法，方法的参数和含义用法如下：

h('div', {class：‘name’}, [
h('p', 'some messages...some messages...'),
h('p', 'some messages...some messages...'),
])

第一个参数：标签
第二个参数：标签属性，比如class，style等
第三个参数：标签的内容，（也可以生成多个标签，或者一个标签的多个值）
